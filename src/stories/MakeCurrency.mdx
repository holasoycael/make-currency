<div className="mb-12">
  <h2 className="text-2xl font-bold mb-4 pb-2 border-b border-slate-100 flex items-center gap-2">⚡ Quick Start</h2>
  
  <h3 className="text-lg font-semibold mt-6 mb-3 text-slate-700">Basic formatting (USD default)</h3>
  
  ```ts
  import { currency } from 'make-currency'
  currency(1234.56)              // "$1,234.56"
  currency(1234.56, { symbol: false }) // "1,234.56"
  currency(0, { isEmpty: true })       // ""
  currency(0, { isEmpty: false })      // "$0.00"
  ```
  
  <h3 className="text-lg font-semibold mt-6 mb-3 text-slate-700">Using a specific currency</h3>
  
  ```ts
  import { currency, TYPES } from 'make-currency'
  currency(1234.56, { money: TYPES.BRL })  // "R$ 1.234,56"
  currency(1234.56, { money: TYPES.EUR })  // "1.234,56 €"
  currency(1234.56, { money: TYPES.JPY })  // "¥1,234.56"
  currency(1234.56, { money: TYPES.GBP })  // "£1,234.56"
  ```
  
  <h3 className="text-lg font-semibold mt-6 mb-3 text-slate-700">Parsing input strings</h3>
  
  ```ts
  import { currencyFn } from 'make-currency'
  currencyFn('123456')   // { floatValue: 1234.56, stringValue: '1,234.56' }
  currencyFn('1')        // { floatValue: 0.01,    stringValue: '0.01' }
  currencyFn('')         // { floatValue: 0,       stringValue: '' }
  ```
  
  <h3 className="text-lg font-semibold mt-6 mb-3 text-slate-700">Global configuration</h3>
  
  ```ts
  import { CONFIGURE, TYPES, currency } from 'make-currency'
  CONFIGURE({ money: TYPES.BRL })
  currency(1234.56)  // "R$ 1.234,56" — BRL is now the default
  ```
</div>
